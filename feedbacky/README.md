## Install

```bash
npm install --save feedbacky
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'feedbacky'
import 'feedbacky/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
